    /**
 * Serviço para captura das informações do prestador através do uso de fontes de dados do Orquestra BPM
 * @see PA-CORP.004 - Consulta Prestador
 * @module PrestadorService
 */

const {
  DATASOURCE_BASE_URL,
  DATASOURCE_REST_GET_PRESTADOR
} = process.env

const ORQUESTRA_REST_GET_PRESTADOR_URL =
  DATASOURCE_BASE_URL + DATASOURCE_REST_GET_PRESTADOR

/**
 * Retorna a CDN do Prestador contratdo
 * @param {object} config - parâmetros de configuração
 * @param {string} config.cpf - CPF do Prestador
 * @param {string|number} config.entidade - código da entidade da RSA
 * @returns {object} CDN do Prestador
 */
export function getContractorCdn ({ cpf, entidade }) {
  const url = new URL(ORQUESTRA_REST_GET_PRESTADOR_URL)

  const params = {
    inpcpf: cpf.replace(/\D/g, ''),
    inpentidade: entidade
  }

  url.search = new URLSearchParams(params).toString()

  return fetch(url)
    .then(res => res.json())
    .then(({ success }) => {
      if (!success.length) return false

      return { cdn: success[0].fields.cdnPrestador }
    })
}
