   * @property {number} workload - carga hor치ria da parcela

  const state = {
    emitter: null,
    installmentType: null,
    expenses: 0,
    service: 0,
    workload: 0,
    total: 0

  const limits = {

      transformer: parseInt,
      value: 0,
      error: false
    },
    service: {
      field: $refs.parcelaServico,
      transformer: currencyToNumber,
      value: 0,
      error: false
    },
    expenses: {
      field: $refs.parcelaDespesas,
      transformer: currencyToNumber,
      value: 0,
      error: false
    },
    total: {
      field: $refs.parcelaTotal,
      transformer: currencyToNumber,
      value: 0,
      error: false
    }
  }

  return {
    mount
  }

  /**

  function mount ({ emitter }) {

      workload: parseInt($refs.parcelaCargaHoraria.value) || 0,
      service: currencyToNumber($refs.parcelaServico.value),
      expenses: currencyToNumber($refs.parcelaDespesas.value),

    state.emitter = emitter

    state.installmentType = getInstallmentType(

  function setup ({ workload, service, expenses, total }) {
    limits.workload.value = workload
    limits.service.value = service
    limits.expenses.value = expenses
    limits.total.value = total
  }

  function addTriggers () {
    [
      $refs.parcelaServico,
      $refs.parcelaDespesas
    ].forEach(field =>
      field.addEventListener('change', setTotal)
    )

    Object.values(limits).forEach(limit =>
      limit.field.addEventListener(
        'change', () => handleLimit(limit)
      )
    )

    $refs.recolheIss
      .addEventListener('change', setTagModifier)
  }

  function setTotal () {
    state.service = currencyToNumber($refs.parcelaServico.value)
    state.expenses = currencyToNumber($refs.parcelaDespesas.value)
    state.workload = parseInt($refs.parcelaCargaHoraria) || 0
    state.total = state.service + state.expenses

    $refs.parcelaTotal.value = numberToCurrency(state.total)

    $refs.parcelaTotal
      .dispatchEvent(new Event('change'))

    state.emitter.emit('installments:values', {
      service: state.service,
      expenses: state.expenses,
      workload: state.workload,
      total: state.total
    })
  }

  function handleLimit (limit) {
    const currentValue = limit.transformer(limit.field.value) || 0
    const hasError = limit.field.value !== '' && currentValue > limit.value

    limit.error = hasError

    if (hasError) {
      addFieldError(limit.field, 'Valor acima do limite m치ximo')
    } else {
      removeFieldError(limit.field)
    }

    checkErrors()
  }

  function checkErrors () {
    const hasErros = Object.values(limits)
      .some(({ error }) => error)

    return hasErros
      ? state.emitter
        .emit('installments:addError', {
          type: 'installment_above_limit',
          message: 'O valor da parcela ultrapassa o valor planejado (limite)'
        })
      : state.emitter
        .emit('installments:removeError', 'installment_above_limit')
  }

  function handleInstallmentType () {
    if (state.installmentType === 'MEDICAO_MENSAL') {
      $refs.parcelaCargaHoraria.setAttribute('required', 'S')

      return $refs.parcelaCargaHoraria
        .closest('.o-form-group')
        .classList
        .remove('hidden')
    }

    if (state.installmentType === 'VALOR_PARCELA') {
      return $refs.parcelaServico
        .closest('.o-form-group')
        .classList
        .remove('hidden')
    }
  }

  function setLabelsHelpers () {
    setFieldHelper(
      $refs.parcelaCargaHoraria,
      `M치x. ${limits.workload.value}h`
    )
    setFieldHelper(
      $refs.parcelaServico,
      `M치x. R$ ${numberToCurrency(limits.service.value)}`
    )
    setFieldHelper(
      $refs.parcelaDespesas,
      `M치x. R$ ${numberToCurrency(limits.expenses.value)}`
    )
  }

  function setTagModifier () {
    const modifier = $refs.recolheIss.value === 'Sim'
      ? '-green'
      : $refs.recolheIss.value === 'N칚o'
        ? '-red'
        : ''

    $refs.recolheIssTag
      .classList.remove('-green', '-red')

    $refs.recolheIssTag
      .classList.add(modifier)
  }
