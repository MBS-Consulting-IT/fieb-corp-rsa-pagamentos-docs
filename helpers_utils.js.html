 * Funções utilitárias para uso geral
 * @module Utils

const currencyFormatter = new Intl.NumberFormat('pt-BR', {
  minimumFractionDigits: 2,
  maximumFractionDigits: 2
})

/**
 * Transforma uma string monetária em número.
 * Quando informado strings vazias o valor 0 é retornado
 * @example '1.000,50' => 1000.5
 * @param {string} currencyValue valor monetário pt-BR
 * @returns {number}
 */
export function currencyToNumber (currencyValue) {
  return parseFloat(
    currencyValue
      .replace(/\./g, '')
      .replace(',', '.') || 0
  )
}

/**
 * Verifica se o valor está no formato monetário
 * @param {string|number} value - valor
 * @returns {boolean}
 */
export function numberToCurrency (number) {
  return currencyFormatter.format(number)
}

/**
 * Altera o valor de um input disparando o evento de `change`
 * @param {HTMLInputElement} input
 * @param {string|number} value valor a ser modificado
 */
export function setValue (input, value) {
  input.value = value
  input.dispatchEvent(new Event('change'))
}

/**
 * Retorna o nome do mês a partir da sua identificação numérica em Javascript (iniciando em 0)
 * @param {string|number} monthNumber - mês em formato numérico, iniciando em 0 (ex. 0 = Janeiro)
 * @param {string} config.month - opção `month` do método `toLocaleString`
 * @param {string} config.lang - opção `lang` do método `toLocaleString`
 * @param {boolean} config.capitalize - capitalizar o mês retornado (ex. Janeiro insted janeiro)
 * @return {string} nome do mês
 */
export function getMonthNameByNumber (monthNumber, config = {
  month: 'long',
  capitalize: true,
  lang: 'pt-BR'
}) {
  const date = new Date()
  const currentMonth = parseInt(monthNumber)

  date.setMonth(currentMonth)

  const month = date.toLocaleString(config.lang, { month: config.month })

  if (!config.capitalize) {
    return month
  }

  return month[0].toUpperCase() + month.slice(1)
}

/**
 * Adiciona meses a uma data
 * @param {Date} date - data a ser manipulada
 * @param {number} quantity - quantidade de meses, podendo ser número negativo para subtração
 * @return {Date} data calculada
 */
export function addMonths (date, quantity) {
  const newDate = new Date(date)
  newDate.setMonth(date.getMonth() + quantity)

  return newDate
}

/**
 * Tranforma uma string de data no formato `dd/mm/yyyy` em objeto de data Javascript
 * @param {string} stringDate - data no formato `dd/mm/yyyy`
 * @return {Date} objeto data
 */
export function parseDate (stringDate) {
  const [dd, mm, yyyy] = stringDate.split('/')
  return new Date(yyyy, mm - 1, dd)
}
